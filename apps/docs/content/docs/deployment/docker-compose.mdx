## Complete Configuration

Create a `docker-compose.yml` file:

```yaml
# docker-compose.yml
version: '3.8'

services:
  postgres:
    image: postgres:16-alpine
    container_name: exlo-postgres
    environment:
      POSTGRES_DB: ${POSTGRES_DB}
      POSTGRES_USER: ${POSTGRES_USER}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
    ports:
      - "${POSTGRES_PORT}:5432"
    volumes:
      - exlo_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER} -d ${POSTGRES_DB}"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - exlo-network

  migrate:
    build:
      context: .
      dockerfile: docker/Dockerfile.migrate
    container_name: exlo-migrate
    environment:
      DATABASE_URL: postgresql://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}
    depends_on:
      postgres:
        condition: service_healthy
    networks:
      - exlo-network
    restart: "no"

  web:
    build:
      context: .
      dockerfile: docker/Dockerfile.web
    container_name: exlo-web
    environment:
      DATABASE_URL: postgresql://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}
      BETTER_AUTH_SECRET: ${BETTER_AUTH_SECRET}
      BETTER_AUTH_URL: ${BETTER_AUTH_URL:-http://localhost:3000}
      INTERNAL_API_SECRET: ${INTERNAL_API_SECRET}
      TUNNL_MANAGEMENT_API_URL: http://tunnl:9090
      NODE_ENV: production
    ports:
      - "${WEB_PORT:-3000}:3000"
    depends_on:
      migrate:
        condition: service_completed_successfully
    restart: unless-stopped
    networks:
      - exlo-network

  tunnl:
    build:
      context: .
      dockerfile: docker/Dockerfile.tunnl
    container_name: exlo-tunnl
    environment:
      DATABASE_URL: postgresql://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}
      API_BASE_URL: ${API_BASE_URL:-http://web:3000}
      INTERNAL_API_SECRET: ${INTERNAL_API_SECRET}
      RUST_LOG: ${RUST_LOG:-info}
    ports:
      - "${SSH_PORT:-2222}:2222"
      - "${PROXY_PORT:-8080}:8080"
    depends_on:
      migrate:
        condition: service_completed_successfully
    restart: unless-stopped
    networks:
      - exlo-network

volumes:
  exlo_data:
    name: exlo_data

networks:
  exlo-network:
    name: exlo-network
    driver: bridge
```

## Environment Variables

### Data Plane (tunnl)

| Variable | Description | Example |
|----------|-------------|---------|
| `DATABASE_URL` | PostgreSQL connection string | `postgres://user:pass@host:5432/db` |
| `API_BASE_URL` | Web service URL for Device Flow | `http://web:3000` |
| `INTERNAL_API_SECRET` | Shared secret for internal API calls | `your-long-random-secret` |
| `RUST_LOG` | Rust log level | `info` |
| `TUNNL_SKIP_AUTH` | Skip Device Flow (dev only) | `true` |

### Control Plane (web)

| Variable | Description | Example |
|----------|-------------|---------|
| `DATABASE_URL` | PostgreSQL connection string | `postgres://user:pass@host:5432/db` |
| `BETTER_AUTH_SECRET` | Better Auth secret (min 32 chars) | `your-auth-secret` |
| `BETTER_AUTH_URL` | Better Auth base URL | `https://tunnl.example.com` |
| `INTERNAL_API_SECRET` | Shared secret for internal API calls | `your-long-random-secret` |
| `TUNNL_MANAGEMENT_API_URL` | Data Plane Management API | `http://tunnl:9090` |

> [!NOTE]
> The Data Plane listens on internal ports `2222/8080/9090`. Use Docker port mappings
> (`SSH_PORT`, `PROXY_PORT`, `WEB_PORT`, `POSTGRES_PORT`) to change host-side exposure.

## Reverse Proxy Configuration

### Nginx

```nginx
# /etc/nginx/sites-available/tunnl

# Web Dashboard
server {
    listen 443 ssl http2;
    server_name tunnl.example.com;

    ssl_certificate /etc/letsencrypt/live/tunnl.example.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/tunnl.example.com/privkey.pem;

    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}

# Tunnel HTTP Proxy (wildcard domain)
server {
    listen 443 ssl http2;
    server_name *.tunnl.example.com;

    ssl_certificate /etc/letsencrypt/live/tunnl.example.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/tunnl.example.com/privkey.pem;

    location / {
        proxy_pass http://127.0.0.1:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSocket support
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
```

### Traefik

Integrate Traefik with Docker Compose:

```yaml
# docker-compose.traefik.yml
version: '3.8'

services:
  traefik:
    image: traefik:v3.2
    command:
      - "--api.insecure=true"
      - "--providers.docker=true"
      - "--providers.docker.exposedbydefault=false"
      - "--entrypoints.web.address=:80"
      - "--entrypoints.websecure.address=:443"
      - "--certificatesresolvers.letsencrypt.acme.tlschallenge=true"
      - "--certificatesresolvers.letsencrypt.acme.email=admin@example.com"
      - "--certificatesresolvers.letsencrypt.acme.storage=/letsencrypt/acme.json"
    ports:
      - "80:80"
      - "443:443"
      - "8080:8080"  # Traefik Dashboard
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - ./letsencrypt:/letsencrypt

  tunnl:
    image: ghcr.io/ZingerLittleBee/EXLO:latest
    ports:
      - "2222:2222"   # SSH Server (exposed directly)
    labels:
      - "traefik.enable=true"
      # Tunnel HTTP Proxy (wildcard domain)
      - "traefik.http.routers.tunnl-proxy.rule=HostRegexp(`{subdomain:[a-z0-9-]+}.tunnl.example.com`)"
      - "traefik.http.routers.tunnl-proxy.entrypoints=websecure"
      - "traefik.http.routers.tunnl-proxy.tls.certresolver=letsencrypt"
      - "traefik.http.routers.tunnl-proxy.tls.domains[0].main=tunnl.example.com"
      - "traefik.http.routers.tunnl-proxy.tls.domains[0].sans=*.tunnl.example.com"
      - "traefik.http.routers.tunnl-proxy.service=tunnl-proxy"
      - "traefik.http.services.tunnl-proxy.loadbalancer.server.port=8080"
    environment:
      - DATABASE_URL=postgres://tunnl:secret@db:5432/tunnl
      - API_BASE_URL=http://web:3000
      - INTERNAL_API_SECRET=your-long-random-secret

  web:
    image: ghcr.io/ZingerLittleBee/EXLO-web:latest
    labels:
      - "traefik.enable=true"
      # Web Dashboard
      - "traefik.http.routers.tunnl-web.rule=Host(`tunnl.example.com`)"
      - "traefik.http.routers.tunnl-web.entrypoints=websecure"
      - "traefik.http.routers.tunnl-web.tls.certresolver=letsencrypt"
      - "traefik.http.routers.tunnl-web.service=tunnl-web"
      - "traefik.http.services.tunnl-web.loadbalancer.server.port=3000"
    environment:
      - DATABASE_URL=postgres://tunnl:secret@db:5432/tunnl
      - INTERNAL_API_SECRET=your-long-random-secret
      - TUNNL_MANAGEMENT_API_URL=http://tunnl:9090

  db:
    image: postgres:16-alpine
    volumes:
      - postgres_data:/var/lib/postgresql/data
    environment:
      - POSTGRES_USER=tunnl
      - POSTGRES_PASSWORD=secret
      - POSTGRES_DB=tunnl

volumes:
  postgres_data:
```

> [!TIP]
> Traefik automatically reads routing configuration from Docker labels and manages HTTPS certificates through Let's Encrypt.

> [!IMPORTANT]
> Wildcard certificates require DNS validation. For Cloudflare, add:
> ```yaml
> - "--certificatesresolvers.letsencrypt.acme.dnschallenge=true"
> - "--certificatesresolvers.letsencrypt.acme.dnschallenge.provider=cloudflare"
> ```

## Start Services

```bash
# Start all services
docker compose up -d

# View logs
docker compose logs -f

# Stop services
docker compose down
```

## Verify Deployment

```bash
# Check SSH service
nc -zv your-domain.com 2222

# Test tunnel connection
ssh -N -R 80:localhost:3000 -p 2222 user@your-domain.com
```
